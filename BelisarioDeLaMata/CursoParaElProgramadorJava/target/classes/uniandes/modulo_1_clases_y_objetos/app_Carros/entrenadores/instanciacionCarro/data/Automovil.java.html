package entrenadores.instanciacionCarro.mundo;

/**
 * Modela el comportamiento de un Automovil: gasolina, color, velocidad y posición
 */
public class Automovil {
    
    //-----------------------------------------------------------
    // Atributos
    //-----------------------------------------------------------
    
    private int velocidad;
    private int gasolina;
    private int colorR;
    private int colorG;
    private int colorB;
    private double distancia;
    
    
    //-----------------------------------------------------------
    // Constructor
    //-----------------------------------------------------------
    
    /**
     * Construye un Automóvil con la cantidad de gasolina especificada
     * 
     * @param gasolinaP - cantidad de gasolina - gasolinaP >= 0
     */
    public Automovil(int gasolinaP)
      {
        gasolina = gasolinaP;
        velocidad = 50;
        distancia = 0;
        colorR = 255;
        colorG = 0;
        colorB = 0;
    }
    
    //-----------------------------------------------------------
    // Métodos
    //-----------------------------------------------------------
    
    
    /**
     * Asigna el color con el que se debe pintar el automóvil.
     * 
     * @param rP - índice de rojo - 0<= rP < 256 
     * @param gP - índice de verde - 0<= gP < 256 
     * @param bP - índice de azul - 0<= bP < 256 
     */
    public void asignarColor(int rP, int gP, int bP) 
      {
        colorR = rP;
        colorG = gP;
        colorB = bP;
    }
    
    
    /**
     * Asigna la velocidad a la que debe ir el automóvil
     * @param velocidadP - velocidad 
     */
    public void asignarVelocidad(int velocidadP) 
      {
        velocidad = velocidadP;
    }


    /**
     * Asigna la gasolina para  el automóvil
     * @param gasolinaP - gasolina 
     */
    public void asignarGasolina(int gasolinaP) 
      {
        gasolina = gasolinaP;
    }


    /**
     * Asigna la distancia para  el automóvil
     * @param distanciaP - distancia 
     */
    public void asignarDistancia(int distanciaP) 
      {
        distancia = distanciaP;
    }


    /**
     * Retorna el color del automovil
     * @return Color 
     */
    public Color darColor() 
      {
        return new Color(colorR, colorG, colorB);
    }


    /**
     * Retorna la cantidad de gasolina del automovil
     * @return int 
     */
    public int darGasolina() 
      {
        return gasolina;
    }


    /**
     * Retorna la velocidad del automovil
     * @return int 
     */
    public int darVelocidad() 
      {
        return velocidad;
    }


    /**
     * Calcula la distancia recorrida por el automovil en una iteracion, teniendo en cuenta su velocidad
     * @return double 
     */
    public double calcularDistanciaAvance() 
      {
        if (gasolina>0)
          {
            distancia= distancia + velocidad;
          }
        return distancia;
    }


    /**
     * Calcula una nueva distancia despues de haber avanzado
     * @return double 
     */
    public double calcularDistanciaDespuesAvance() 
      {
        distancia = distancia - velocidad;
        return distancia;
    }


    /**
     * Calcula la gasolina en un momento dado teniendo cuenta el avance
     * @return double 
     */
    public int calcularGasolinaDespuesAvance() 
      {
        if (gasolina > 0)
          {
            gasolina = gasolina - Math.abs(velocidad);
        }
          else
          {
            gasolina = 0;
        }
        return distancia;
    }

}
