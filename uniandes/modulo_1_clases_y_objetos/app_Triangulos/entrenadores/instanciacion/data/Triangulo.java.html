package entrenadores.instanciacion.example;


/**
 * Modela las características geométricas de un 
 * triángulo además del color con el que se debe
 * pintar el relleno y el borde.
 *  
 * @author c-noguera
 */
public class Triangulo {
    
    //---------------------------------------------------
    //  Atributos
    //---------------------------------------------------
    
    private int x1;
    private int y1;
    
    private int x2;
    private int y2;
    
    private int x3;
    private int y3;
    
    private int rellenoR;
    private int rellenoG;
    private int rellenoB;
    
    private int bordeR;
    private int bordeG;
    private int bordeB;
    
    //--------------------------------------------------
    // Constructoras
    //--------------------------------------------------

    public Triangulo(  int x1P, int y1P,
                  int x2P, int y2P,
                  int x3P, int y3P){
        this.x1 = x1P;
        this.y1 = y1P;
        
        this.x2 = x2P;
        this.y2 = y2P;
        
        this.x3 = x3P;
        this.y3 = y3P;
    }
    
    //-------------------------------------------------
    // Métodos
    //-------------------------------------------------
    
    
   /**
    * Mueve el primer vértice del triángulo a la posición especificada <br>
    * <b>pre:</b>true  <br>
    * 
    * @param xP - coordenada en x de la nueva posicion del primer vértice - xP > 0
    * @param yP - coordenada en y de la nueva posicion del primer vértice - yP > 0
    * 
    */ 
    public void moverP1(int xP, int yP){
        x1 = xP;
        y1 = yP;
    }
    
    
    /**
     * Mueve el segundo vértice del triángulo a la posición especificada <br>
     * <b>pre:</b> true <br>
     * 
     * @param xP - coordenada en x de la nueva posicion del segundo vértice - xP > 0
     * @param yP - coordenada en y de la nueva posicion del segundo vértice - yP > 0
     * 
     */
    public void moverP2(int xP, int yP){
        x2 = xP;
        y2 = xP;
    }
    
    
    /**
     * Mueve el tercer vértice del triángulo a la posición especificada <br>
     * <b>pre:</b> true > 0 <br>
     * 
     * @param xP - coordenada en x de la nueva posicion del tercer vértice - xP > 0
     * @param yP - coordenada en y de la nueva posicion del tercer vértice - yP > 0
     * 
     */
    public void moverP3(int xP, int yP){
        x3 = xP;
        y3 = yP;
    }
    
    
    /**
     * Cambia el color con el que se rellena el interior del triángulo<br>
     * <b>pre:</b> true <br>
     * 
     * @param rP - índice de rojo - 0<= rP < 256 
     * @param gP - índice de verde - 0<= gP < 256 
     * @param bP - índice de azul - 0<= bP < 256 
     */
    public void definirColorRelleno(int rP, int gP, int bP){
        rellenoR = rP;
        rellenoG = gP;
        rellenoB = bP;
    }
    
    /**
     * Cambia el color con el que se dibuja el borde del triángulo<br>
     * <b>pre:</b> 0<= rP, gP, bP < 256 <br>
     * 
     * @param rP - índice de rojo - 0<= rP < 256
     * @param gP - índice de verde - 0<= gP < 256
     * @param bP - índice de azul - 0<= bP < 256
     */   
    public void definirColorBorde(int rP, int gP, int bP){
        bordeR = rP;
        bordeG = gP;
        bordeB = bP;
    }
}
